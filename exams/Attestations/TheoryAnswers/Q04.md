### 4. Очередь с приоритетом. Операции. Бинарная куча (min) и её свойство
Очередь с приоритетом хранит элементы и умеет: `Insert`, `Peek` (минимум), `Extract` (удалить минимум), часто `DecreaseKey`. Min‑heap — почти полное бинарное дерево, хранимое в массиве; индексация потомков `2i+1/2i+2`, предка `(i-1)/2`. Свойство: ключ в вершине ≤ ключей в детях, поэтому минимум всегда в корне.  
```csharp
class MinHeap{
    private readonly List<int> h=new();
    void SiftUp(int i){ while(i>0){int p=(i-1)/2; if(h[p]<=h[i]) break; (h[p],h[i])=(h[i],h[p]); i=p;} }
    void SiftDown(int i){ while(true){int l=2*i+1,r=l+1,b=i; if(l<h.Count && h[l]<h[b]) b=l; if(r<h.Count && h[r]<h[b]) b=r; if(b==i) break; (h[b],h[i])=(h[i],h[b]); i=b;} }
    public void Insert(int x){ h.Add(x); SiftUp(h.Count-1); }
    public int Peek()=>h[0];
    public int Extract(){ int res=h[0]; h[0]=h[^1]; h.RemoveAt(h.Count-1); if(h.Count>0) SiftDown(0); return res; }
}
```
